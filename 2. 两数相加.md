提示：本意是要求你**模拟大数加法**

```c++
class Solution {
public:
    void update(int& sum, int& prev) {
        if (sum >= 10) { // 题目确保只是单个数的相加
            prev = 1;
            sum = sum - 10;
        } else {
            prev = 0;
        }
    }
    ListNode* addTwoNumbers(ListNode* l1, ListNode* l2) {
        auto virtual_head = new ListNode(-1);
        auto record = virtual_head;
        int prev = 0;
        while (l1 || l2 || prev) {
            int n1 = l1 ? l1->val : 0;
            int n2 = l2 ? l2->val : 0;

            int sum = n1 + n2 + prev;
            update(sum, prev);
            auto node = new ListNode(sum);
            record->next = node;
            
            record = record->next;
            if (l1) l1 = l1->next;
            if (l2) l2 = l2->next;
        }
        return virtual_head->next;
    }
};
```
